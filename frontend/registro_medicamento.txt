// Cargar pacientes
async function cargarPacientes() {
  const selectorPaciente = document.getElementById("paciente");

  // Estado inicial mientras carga
  selectorPaciente.innerHTML = '<option value="">Cargando pacientes...</option>';
  selectorPaciente.disabled = true;

  try {
    const respuesta = await fetch(`${BASE_URL}/pacientes`);

    if (!respuesta.ok) {
      throw new Error("No se pudieron obtener los pacientes");
    }

    const datos = await respuesta.json();

    // Ajusta esta línea si tu backend devuelve otra estructura
    const pacientes = datos.pacientes || datos;

    if (!pacientes || pacientes.length === 0) {
      selectorPaciente.innerHTML = '<option value="">No hay pacientes registrados</option>';
      return;
    }

    selectorPaciente.innerHTML = '<option value="">Seleccione un paciente</option>';

    pacientes.forEach((paciente) => {
      const opcion = document.createElement("option");
      opcion.value = paciente.id;
      opcion.textContent = `${paciente.nombres} ${paciente.apellidos} - ${paciente.numero_documento}`;
      selectorPaciente.appendChild(opcion);
    });

    selectorPaciente.disabled = false;

  } catch (error) {
    console.error("Error al cargar pacientes:", error);
    selectorPaciente.innerHTML = '<option value="">Error al cargar pacientes</option>';
  }
}


  // Creamos los botones de dias de la semana
  var nombresDias  = ['Lun', 'Mar', 'Mie', 'Jue', 'Vie', 'Sab', 'Dom'];
  var clavesDias   = ['lun', 'mar', 'mie', 'jue', 'vie', 'sab', 'dom'];
  var grupoDias = document.getElementById('grupoDias');

  for (var j = 0; j < nombresDias.length; j++) {
    var boton = document.createElement('button');
    boton.type = 'button';
    boton.className = 'boton-dia';
    boton.textContent = nombresDias[j];
    boton.dataset.clave = clavesDias[j];
    boton.onclick = function () {
      // Al hacer clic alternamos el dia entre seleccionado y no seleccionado
      if (this.classList.contains('seleccionado')) {
        this.classList.remove('seleccionado');
        listaDias = listaDias.filter(function(d) { return d !== this.dataset.clave; }.bind(this));
      } else {
        this.classList.add('seleccionado');
        listaDias.push(this.dataset.clave);
      }
    };
    grupoDias.appendChild(boton);
  }


  // Cuando cambia la frecuencia, mostramos u ocultamos la seccion de dias
  document.getElementById('frecuencia').addEventListener('change', function () {
    var frecuenciasConDias = ['dias_alternos', 'semanal', 'personalizado'];
    if (frecuenciasConDias.indexOf(this.value) !== -1) {
      seccionDias.style.display = 'block';
    } else {
      seccionDias.style.display = 'none';
      // Si ocultamos la seccion, limpiamos los dias seleccionados
      listaDias = [];
      var botonesDia = document.querySelectorAll('.boton-dia');

  // Agrega una hora como pastilla al contenedor de horarios
  function agregarHorario() {
    var hora = inputHora.value;

    if (!hora) {
      inputHora.focus();
      return;
    }

    // No permitimos horarios duplicados
    if (listaHorarios.indexOf(hora) !== -1) {
      mostrarNotificacion('Ese horario ya fue agregado.', true);
      inputHora.value = '';
      return;
    }

    // Agregamos la hora a la lista
    listaHorarios.push(hora);

    // Creamos la pastilla visual con la hora y un boton para quitarla
    var pastilla = document.createElement('span');
    pastilla.className = 'pastilla';
    pastilla.innerHTML = hora + ' <button type="button" onclick="quitarHorario(\'' + hora + '\', this)">x</button>';
    cajaHorarios.insertBefore(pastilla, inputHora);

    // Limpiamos el input para ingresar otro horario
    inputHora.value = '';

    // Marcamos la caja como correcta y ocultamos el error
    cajaHorarios.classList.remove('error');
    cajaHorarios.classList.add('correcto');
    document.getElementById('error-horarios').classList.remove('visible');
  }

  // Quita un horario cuando el usuario presiona la X de la pastilla
  function quitarHorario(hora, boton) {
    listaHorarios = listaHorarios.filter(function(h) { return h !== hora; });
    boton.parentElement.remove();
    if (listaHorarios.length === 0) {
      cajaHorarios.classList.remove('correcto');
    }
  }

  // Presionar Enter en el input de hora tambien agrega el horario
  inputHora.addEventListener('keydown', function(evento) {
    if (evento.key === 'Enter') {
      evento.preventDefault();
      agregarHorario();
    }
  });

  // Valida un campo: lo marca verde si esta correcto, rojo si tiene error
  // Devuelve true si esta bien, false si no
  function validarCampo(campo, idMensajeError) {
    campo.classList.remove('error', 'correcto');
    var valor = campo.value.trim();
    var tieneError = false;

    if (!valor) {
      tieneError = true;
    }
    if (campo.type === 'number' && parseFloat(valor) <= 0) {
      tieneError = true;
    }

    if (tieneError) {
      campo.classList.add('error');
      document.getElementById(idMensajeError).classList.add('visible');
    } else {
      campo.classList.add('correcto');
      document.getElementById(idMensajeError).classList.remove('visible');
    }

    return !tieneError;
  }

  // Si el input trae 2026-03-28 la funcion lo convierte a 03/28/2026
  function convertirFechaAFormatoBackend(fechaISO) {
  if (!fechaISO) return "";

  var partes = fechaISO.split("-");
  if (partes.length !== 3) return fechaISO;

  return partes[1] + "/" + partes[2] + "/" + partes[0];

function limpiarFormulario() {
  formulario.reset();

  // Quitamos los colores de validacion de todos los campos
  var todosCampos = formulario.querySelectorAll('.form-control, .form-select');
  for (var i = 0; i < todosCampos.length; i++) {
    todosCampos[i].classList.remove('correcto', 'error');
  }

  // Borramos todas las pastillas de horario
  var pastillas = document.querySelectorAll('.pastilla');
  for (var j = 0; j < pastillas.length; j++) {
    pastillas[j].remove();
  }
  listaHorarios = [];
  cajaHorarios.classList.remove('correcto', 'error');

  // Desmarcamos todos los dias
  var botonesDia = document.querySelectorAll('.boton-dia');
  for (var k = 0; k < botonesDia.length; k++) {
    botonesDia[k].classList.remove('seleccionado');
  }
  listaDias = [];

  // Ocultamos la seccion de dias
  seccionDias.style.display = 'none';

  // Reiniciamos el contador de instrucciones
  document.getElementById('contadorLetras').textContent = '0';

  // Volvemos a poner la fecha de hoy
  document.getElementById('fechaInicio').value = fechaHoy;

  // Ocultamos la alerta de error
  alertaError.classList.remove('visible');
}
